2. Secure Coding Practices

Input Validation and Sanitization

Proper input validation is the first line of defense against injection attacks. Always validate, sanitize, and escape user inputs. 

// JavaScript - Input Validation Example
function validateEmail(email) {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
        throw new Error('Invalid email format');
    }
    return email.trim().toLowerCase();
}

function sanitizeInput(input) {
    // Remove potentially dangerous characters
    return input.replace(/[<>"'&]/g, function(match) {
        const escapeMap = {
            '<': '<',
            '>': '>',
            '"': '"',
            "'": ''',
            '&': '&'
        };
        return escapeMap[match];
    });
}

Authentication and Authorization

Implement robust authentication mechanisms and proper authorization controls to protect sensitive resources. 

// Node.js - JWT Authentication with bcrypt
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');

// Password hashing
async function hashPassword(password) {
    const saltRounds = 12;
    return await bcrypt.hash(password, saltRounds);
}

// Password verification
async function verifyPassword(password, hashedPassword) {
    return await bcrypt.compare(password, hashedPassword);
}

// JWT token generation
function generateToken(user) {
    return jwt.sign(
        { 
            userId: user.id, 
            email: user.email,
            role: user.role 
        },
        process.env.JWT_SECRET,
        { 
            expiresIn: '1h',
            issuer: 'your-app-name',
            audience: 'your-app-users'
        }
    );
}

// Middleware for token verification
function authenticateToken(req, res, next) {
    const authHeader = req.headers['authorization'];
    const token = authHeader && authHeader.split(' ')[1];
    
    if (!token) {
        return res.status(401).json({ error: 'Access token required' });
    }
    
    jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
        if (err) {
            return res.status(403).json({ error: 'Invalid or expired token' });
        }
        req.user = user;
        next();
    });
}

const userSchema = new mongoose.Schema({
    email: {
        type: String,
        required: true,
        unique: true,
        validate: {
            validator: function(v) {
                return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v);
            },
            message: 'Invalid email format'
        }
    },
    password: {
        type: String,
        required: true,
        minlength: 8
    },
    personalData: {
        type: String,
        // Field-level encryption for sensitive data
        encrypt: true
    },
    loginAttempts: {
        type: Number,
        default: 0
    },
    lockUntil: Date
});

// Account lockout mechanism
userSchema.methods.incLoginAttempts = function() {
    if (this.lockUntil && this.lockUntil < Date.now()) {
        return this.updateOne({
            $unset: { lockUntil: 1 },
            $set: { loginAttempts: 1 }
        });
    }
    
    const updates = { $inc: { loginAttempts: 1 } };
    
    if (this.loginAttempts + 1 >= 5 && !this.lockUntil) {
        updates.$set = {
            lockUntil: Date.now() + 2 * 60 * 60 * 1000 // 2 hours
        };
    }
    
    return this.updateOne(updates);
};

5. Identity & Access Management (IAM)

Multi-Factor Authentication (MFA)

Implement robust MFA using TOTP, SMS, or hardware tokens to add an extra layer of security. 

// Node.js - TOTP Implementation
const speakeasy = require('speakeasy');
const QRCode = require('qrcode');

class MFAService {
    // Generate secret for new user
    static generateSecret(userEmail) {
        const secret = speakeasy.generateSecret({
            name: `SecureApp (${userEmail})`,
            issuer: 'SecureApp',
            length: 32
        });
        
        return {
            secret: secret.base32,
            qrCodeUrl: secret.otpauth_url
        };
    }
    
    // Generate QR code for mobile app setup
    static async generateQRCode(otpauthUrl) {
        try {
            const qrCodeDataUrl = await QRCode.toDataURL(otpauthUrl);
            return qrCodeDataUrl;
        } catch (error) {
            throw new Error('Failed to generate QR code');
        }
    }
    
    // Verify TOTP token
    static verifyToken(token, secret) {
        return speakeasy.totp.verify({
            secret: secret,
            encoding: 'base32',
            token: token,
            window: 2 // Allow 2 time steps (60 seconds) tolerance
        });
    }
    
    // Backup codes generation
    static generateBackupCodes() {
        const codes = [];
        for (let i = 0; i < 10; i++) {
            const code = Math.random().toString(36).substring(2, 10).toUpperCase();
            codes.push(code);
        }
        return codes;
    }
}

// MFA middleware
function requireMFA(req, res, next) {
    const { mfaToken } = req.body;
    const user = req.user;
    
    if (!user.mfaEnabled) {
        return next();
    }
    
    if (!mfaToken) {
        return res.status(401).json({ 
            error: 'MFA token required',
            mfaRequired: true 
        });
    }
    
    const isValid = MFAService.verifyToken(mfaToken, user.mfaSecret);
    
    if (!isValid) {
        return res.status(401).json({ error: 'Invalid MFA token' });
    }
    
    next();
}

Role-Based Access Control (RBAC)

Implement fine-grained access control with roles and permissions to ensure users only access what they need. 

// RBAC Implementation
class RBACService {
    constructor() {
        this.roles = new Map();
        this.permissions = new Map();
        this.setupDefaultRoles();
    }
    
    setupDefaultRoles() {
        // Define permissions
        this.permissions.set('read:users', 'Read user information');
        this.permissions.set('write:users', 'Create and update users');
        this.permissions.set('delete:users', 'Delete users');
        this.permissions.set('read:admin', 'Read admin panel');
        this.permissions.set('write:admin', 'Admin panel modifications');
        
        // Define roles with permissions
        this.roles.set('user', {
            name: 'User',
            permissions: ['read:users']
        });
        
        this.roles.set('moderator', {
            name: 'Moderator',
            permissions: ['read:users', 'write:users']
        });
        
        this.roles.set('admin', {
            name: 'Administrator',
            permissions: [
                'read:users', 'write:users', 'delete:users',
                'read:admin', 'write:admin'
            ]
        });
    }
    
    hasPermission(userRole, requiredPermission) {
        const role = this.roles.get(userRole);
        if (!role) return false;
        
        return role.permissions.includes(requiredPermission);
    }
    
    hasAnyPermission(userRole, requiredPermissions) {
        return requiredPermissions.some(permission => 
            this.hasPermission(userRole, permission)
        );
    }
    
    hasAllPermissions(userRole, requiredPermissions) {
        return requiredPermissions.every(permission => 
            this.hasPermission(userRole, permission)
        );
    }
}

const rbac = new RBACService();

// Middleware for permission checking
function requirePermission(permission) {
    return (req, res, next) => {
        const user = req.user;
        
        if (!user) {
            return res.status(401).json({ error: 'Authentication required' });
        }
        
        if (!rbac.hasPermission(user.role, permission)) {
            return res.status(403).json({ 
                error: 'Insufficient permissions',
                required: permission,
                userRole: user.role
            });
        }
        
        next();
    };
}

Security Monitoring and Logging

Implement comprehensive logging and monitoring to detect suspicious activities and security incidents. 

// Security Event Logging System
const winston = require('winston');
const { ElasticsearchTransport } = require('winston-elasticsearch');

class SecurityLogger {
    constructor() {
        this.logger = winston.createLogger({
            level: 'info',
            format: winston.format.combine(
                winston.format.timestamp(),
                winston.format.errors({ stack: true }),
                winston.format.json()
            ),
            transports: [
                new winston.transports.File({ 
                    filename: 'logs/security.log',
                    level: 'warn'
                }),
                new winston.transports.Console({
                    format: winston.format.simple()
                }),
                new ElasticsearchTransport({
                    level: 'info',
                    clientOpts: { node: process.env.ELASTICSEARCH_URL },
                    index: 'security-logs'
                })
            ]
        });
    }
    
    logSecurityEvent(event, details = {}) {
        const securityEvent = {
            type: 'SECURITY_EVENT',
            event: event,
            timestamp: new Date().toISOString(),
            ip: details.ip,
            userAgent: details.userAgent,
            userId: details.userId,
            sessionId: details.sessionId,
            severity: details.severity || 'medium',
            details: details.additionalInfo || {}
        };
        
        this.logger.warn('Security Event', securityEvent);
        
        // Trigger alerts for high severity events
        if (details.severity === 'high') {
            this.triggerAlert(securityEvent);
        }
    }
    
    logFailedLogin(ip, userAgent, email, reason) {
        this.logSecurityEvent('FAILED_LOGIN', {
            ip,
            userAgent,
            email,
            reason,
            severity: 'medium',
            additionalInfo: { loginAttempt: true }
        });
    }
    
    logSuspiciousActivity(ip, userAgent, activity, userId = null) {
        this.logSecurityEvent('SUSPICIOUS_ACTIVITY', {
            ip,
            userAgent,
            userId,
            activity,
            severity: 'high',
            additionalInfo: { requiresInvestigation: true }
        });
    }
    
    async triggerAlert(event) {
        // Send to security team via Slack/email
        const alertMessage = {
            text: `ðŸš¨ Security Alert: ${event.event}`,
            attachments: [{
                color: 'danger',
                fields: [
                    { title: 'Event', value: event.event, short: true },
                    { title: 'Severity', value: event.severity, short: true },
                    { title: 'IP Address', value: event.ip, short: true },
                    { title: 'User ID', value: event.userId || 'N/A', short: true },
                    { title: 'Timestamp', value: event.timestamp, short: false }
                ]
            }]
        };
        
        // Send to Slack webhook
        await fetch(process.env.SLACK_WEBHOOK_URL, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(alertMessage)
        });
    }
}

const securityLogger = new SecurityLogger();

// Middleware for security logging
function securityLoggingMiddleware(req, res, next) {
    const originalSend = res.send;
    
    res.send = function(data) {
        // Log failed authentication attempts
        if (res.statusCode === 401) {
            securityLogger.logFailedLogin(
                req.ip,
                req.get('User-Agent'),
                req.body.email,
                'Invalid credentials'
            );
        }
        
        // Log suspicious 4xx errors
        if (res.statusCode >= 400 && res.statusCode < 500) {
            securityLogger.logSuspiciousActivity(
                req.ip,
                req.get('User-Agent'),
                `${req.method} ${req.path} - ${res.statusCode}`,
                req.user?.id
            );
        }
        
        originalSend.call(this, data);
    };
    
    next();
}
